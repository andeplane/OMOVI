## Install

```bash
npm install --save omovi
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'omovi'
import 'omovi/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
